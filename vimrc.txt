function! InsertSpecLine()
  " Yank current function name
  normal! yiw
  let fname = @"
  " Insert @spec line above with function name
  execute "normal! O@spec " . fname . "() :: "
endfunction

nnoremap <Leader>s :call InsertSpecLine()<CR>

function! WrapWithDefBlock()
  " Yank the current word under the cursor
  normal! yiw
  let fname = @"
  " Replace the word with def ... do ... end
  let replacement = "def " . fname . "() do\nend"
  " Visually select the word and paste the block
  normal! viw"_d
  call append(line('.') - 1, split(replacement, '\n'))
endfunction

" Press , then f
nnoremap <Leader>f :call WrapWithDefBlock()<CR>

function! RunCurrentTestInTmuxWindow()
  let filepath = expand('%:p')        " full path to current file
  let line = line('.')
  let cmd = 'mix test ' . filepath . ':' . line

  " Run in new tmux window and keep it open
  let tmux_cmd = 'tmux new-window -n test_runner "clear && echo Running: ' . cmd . ' && ' . cmd . ' ; exec $SHELL"'

  call system(tmux_cmd)
endfunction

nnoremap <Leader>t :call RunCurrentTestInTmuxWindow()<CR>

function! InsertModuleBlock()
  " Get module name under cursor
  normal! yiw
  let modname = @"

  " Delete the current line (assumes it's just the module name)
  let line_num = line('.')
  execute line_num . 'delete _'

  " Insert the module block in its place
  let lines = [
        \ 'defmodule ' . modname . ' do',
        \ '  @moduledoc false',
        \ 'end'
        \ ]
  call append(line_num - 1, lines)

  " Auto-indent the newly inserted block
  execute line_num . ',' . (line_num + len(lines) - 1) . 'normal ='
endfunction

nnoremap <Leader>mb :call InsertModuleBlock()<CR>

function! ShowBranchCommitsInBuffer()
  let lines = systemlist('git log origin/main..HEAD --pretty=format:"%ar %an %s"')

  new
  call setline(1, lines)
  setlocal buftype=nofile bufhidden=wipe nobuflisted noswapfile readonly

  " Syntax highlighting
  syntax match GitTime /^\S\+\s\+\S\+/
  syntax match GitAuthor /^\S\+\s\+\zs\S\+\s\+\S\+/
  syntax match GitMessage /^\S\+\s\+\S\+\s\+\S\+\s\+\zs.*/

  highlight GitTime ctermfg=Yellow guifg=Yellow
  highlight GitAuthor ctermfg=Cyan guifg=Cyan
  highlight GitMessage ctermfg=White guifg=White
endfunction

nnoremap <Leader>g :call ShowBranchCommitsInBuffer()<CR>
